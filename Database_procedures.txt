-- Example 1: Single Parameter
CREATE PROCEDURE firstParamProcedure(@choiceAge as Integer)
AS
BEGIN
SELECT 
Name,  Nationality, Age
FROM Pilot
WHERE
Age <= @choiceAge 
ORDER BY 
Name;
END
GO

EXEC firstParamProcedure  @choiceAge =4;





-- Example 2: Multiple parameters
CREATE PROCEDURE MoreParam(@choiceAge as Integer, @citizen as
Varchar)
AS
BEGIN
SELECT
Name, Nationality, Age
FROM Pilot
WHERE
Age <= @choiceAge AND Nationality=@citizen
ORDER BY
Name;
END
GO

EXEC MoreParam @choiceAge=45, @citizen='Ghanaian';

@PID int
AS
BEGIN
